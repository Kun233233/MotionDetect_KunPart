#include <stdio.h>
#include <OpenNI.h>
#include <iostream>

#include "OniSampleUtilities.h"

#define SAMPLE_READ_WAIT_TIMEOUT 2000 //2000ms
const int MAX_RECORD_FRAME_NUM = 100;

using namespace openni;

int main(int argc, char* argv[])
{
	//initialize openni sdk
	Status rc = OpenNI::initialize();
	if (rc != STATUS_OK)
	{
		printf("Initialize failed\n%s\n", OpenNI::getExtendedError());
		return 1;
	}

	Device device;

	//open device
	rc = device.open(ANY_DEVICE);
	if (rc != STATUS_OK)
	{
		printf("Couldn't open device\n%s\n", OpenNI::getExtendedError());
		return 2;
	}

	VideoStream depth;

	//create depth stream
	if (device.getSensorInfo(SENSOR_DEPTH) != NULL)
	{
		rc = depth.create(device, SENSOR_DEPTH);
		if (rc != STATUS_OK)
		{
			printf("Couldn't create depth stream\n%s\n", OpenNI::getExtendedError());
			return 3;
		}
	}

	//start depth stream
	rc = depth.start();
	if (rc != STATUS_OK)
	{
		printf("Couldn't start the depth stream\n%s\n", OpenNI::getExtendedError());
		return 4;
	}

	// -------------- record preparation -----------------
	//定义一个录制器对象
	Recorder recordDepth;

	char depthPathFile[256] = "depth.ppm";//irPathFile为oni文件保存的路径

	//创建录制器，需要一个oni文件路径作为参数
	rc = recordDepth.create(depthPathFile);
	if (STATUS_OK != rc)
	{
		std::cout << "create ir video recorder failed." << std::endl;
		return rc;
	}

	//将IR流添加到录制器上
	rc = recordDepth.attach(depth);
	if (STATUS_OK != rc)
	{
		std::cout << "attach ir video recorder failed." << std::endl;
		return rc;
	}

	//启动录制器
	rc = recordDepth.start();
	if (STATUS_OK != rc)
	{
		std::cout << "start ir video recorder failed." << std::endl;
		return rc;
	}
	// ---------------------- end --------------------------






	VideoFrameRef frame;

	int count = 0;

	while (count < MAX_RECORD_FRAME_NUM)
	{
		int changedStreamDummy;
		VideoStream* pStream = &depth;

		//wait a new frame
		rc = OpenNI::waitForAnyStream(&pStream, 1, &changedStreamDummy, SAMPLE_READ_WAIT_TIMEOUT);
		if (rc != STATUS_OK)
		{
			printf("Wait failed! (timeout is %d ms)\n%s\n", SAMPLE_READ_WAIT_TIMEOUT, OpenNI::getExtendedError());
			continue;
		}

		//get depth frame
		rc = depth.readFrame(&frame);
		if (rc != STATUS_OK)
		{
			printf("Read failed!\n%s\n", OpenNI::getExtendedError());
			continue;
		}

		//check if the frame format is depth frame format
		if (frame.getVideoMode().getPixelFormat() != PIXEL_FORMAT_DEPTH_1_MM && frame.getVideoMode().getPixelFormat() != PIXEL_FORMAT_DEPTH_100_UM)
		{
			printf("Unexpected frame format\n");
			continue;
		}

		DepthPixel* pDepth = (DepthPixel*)frame.getData();

		int middleIndex = (frame.getHeight() + 1)*frame.getWidth() / 2;

		//print the depth value of the middle pixel of the frame
		printf("[%08llu] %8d\n", (long long)frame.getTimestamp(), pDepth[middleIndex]);

		count++;
	}


	//关闭及销毁录制器
	recordDepth.stop();
	recordDepth.destroy();

	//stop depth stream
	depth.stop();

	//destroy depth stream
	depth.destroy();

	//close device
	device.close();

	//shutdown OpenNI
	OpenNI::shutdown();

	return 0;
}
